The arguments to the following MathML constructs must have a numeric type:
\token{plus}, \token{minus}, \token{times}, \token{divide}, \token{power},
\token{root}, \token{abs}, \token{exp}, \token{ln}, \token{log},
\token{floor}, \token{ceiling}, \token{factorial}, \token{sin},
\token{cos}, \token{tan}, \token{sec}, \token{csc}, \token{cot},
\token{sinh}, \token{cosh}, \token{tanh}, \token{sech}, \token{csch},
\token{coth}, \token{arcsin}, \token{arccos}, \token{arctan},
\token{arcsec}, \token{arccsc}, \token{arccot}, \token{arcsinh},
\token{arccosh}, \token{arctanh}, \token{arcsech}, \token{arccsch},
\token{arccoth}.  (References: L2V2 Section~3.5.8;
\changed{L2V3 Section~\ref{sec:mathmltype}.})
